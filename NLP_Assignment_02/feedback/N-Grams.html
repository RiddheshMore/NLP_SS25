 35    def window(self, seq, n=2):
 36        '''
 37        Returns a sliding window (of width n) over data from the iterable

 57        result = tuple(islice(it, n))
 58        if len(result) == n:
 59            yield result
 60        for elem in it:
 61            result = result[1:] + (elem,)
